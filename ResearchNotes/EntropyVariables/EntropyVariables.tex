The Navier-Stokes equations are
\begin{equation}
\Divxt\vecttwo{\bfF}{\bfU}=\bs{\mathcal{F}}
\end{equation}
where
\begin{equation*}
\bfU=\vectthree{U_c}{\bfU_m}{U_e}=\rho\vectthree{1}{\bfu}{e_0}
\end{equation*}
and
\begin{align*}
\bfF
&=\arrthreeone
{\bfF_1-\sum_{j=1}^dK_{1j}\bfU_{,j}}
{\bfF_2-\sum_{j=1}^dK_{2j}\bfU_{,j}}
{\bfF_3-\sum_{j=1}^dK_{3j}\bfU_{,j}}
% &=\vectthree{\rho\bfu}
% {\rho\bfu\otimes\bfu+\frac{R}{C_v}(\rho e_0-\frac{1}{2}\rho\bfu\cdot\bfu)\bfI-\mathbb{D}}
% {\rho\bfu e_0+\frac{R}{C_v}(\rho e_0-\frac{1}{2}\rho\bfu\cdot\bfu)\bfu+\bfq-\bfu\cdot\mathbb{D}}\\
% &=\vectthree{\bfU_m}{\bfU_m\otimes\bfU_m/U_c+\frac{R}{C_v}(U_e-\frac{1}{2}\bfU_m\cdot\bfU_m/U_c)\bfI-\mathbb{D}}
% {\bfU_mU_e/U_c+\frac{R}{C_v}(U_e-\frac{1}{2}\bfU_m\cdot\bfU_m/U_c)\bfU_m/U_c+\bfq-\bfU_m/U_c\cdot\mathbb{D}}
\end{align*}

Multiplying by test function $\bfW$ and integrating by parts, we get
\begin{equation}
-\LRp{\vecttwo{\bfF}{\bfU},\Gradxt \bfW}+\LRa{\hat{\bfT}, \bfW}=\LRp{\bs{\mathcal{F}},\bfW}
\end{equation}
Assuming summation on $i$, we can also write this
\begin{equation}
-\LRp{\bfU,\bfW_{,t}}-\LRp{\bfF_i-\sum_{j=1}^dK_{ij}\bfU_{,j},\bfW_{,i}}+\LRa{\hat{\bfT}, \bfW}=\LRp{\bs{\mathcal{F}},\bfW}
\end{equation}
